1141 template<class R> inline std::pair<std::vector<future<R>>, std::vector<async_io_op>> async_file_io_dispatcher_base::call(const std::vector<async_io_op> &ops, const std::vector<std::function<R()>> &callables)
1142 {
1143  typedef packaged_task<R()> tasktype;
1144  std::vector<future<R>> retfutures;
1145  std::vector<std::pair<async_op_flags, std::function<completion_t>>> callbacks;
1146  retfutures.reserve(callables.size());
1147  callbacks.reserve(callables.size());
1148  auto f=[](size_t, std::shared_ptr<detail::async_io_handle> _, std::shared_ptr<tasktype> c) {

1151  };
1152  for(auto &t : callables)
1153  {
1154  std::shared_ptr<tasktype> c(std::make_shared<tasktype>(t));
1155  retfutures.push_back(c->get_future());
1156  callbacks.push_back(std::make_pair(async_op_flags::None, std::bind(f, std::placeholders::_1, std::placeholders::_2, std::move(c))));
1157  }
